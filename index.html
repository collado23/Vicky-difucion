function recordVideo() { const canvas = document.getElementById('render-canvas'); const stream = canvas.captureStream(30); const recorder = new MediaRecorder(stream, { mimeType: 'video/webm' }); let chunks = []; recorder.ondataavailable = e => chunks.push(e.data); recorder.onstop = () => { const blob = new Blob(chunks, { type: 'video/webm' }); const url = URL.createObjectURL(blob); // FORZAMOS LA DESCARGA CON UN ENLACE MANUAL const a = document.createElement('a'); a.style.display = 'block'; a.href = url; a.download = 'mi_video_umbanda.webm'; a.innerHTML = "⬇️ HACER CLIC AQUÍ PARA DESCARGAR TU VIDEO"; a.style.padding = "20px"; a.style.background = "#00ff00"; a.style.color = "#000"; a.style.marginTop = "10px"; a.style.textDecoration = "none"; a.style.fontWeight = "bold"; document.body.appendChild(a); // Aparecerá un botón verde gigante debajo }; recorder.start(); alert("Grabando... Espera 10 segundos a que termine."); setTimeout(() => recorder.stop(), 10000); }
